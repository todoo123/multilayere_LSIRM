%% ============================================================
\section{Notation}
%% ============================================================

\begin{itemize}
  \item $n$: number of respondents (subjects), indexed by $i = 1,\dots,n$.
  \item $P_l$: number of items in layer $l$ ($l=1,\dots,5$).
  \item $d$: dimension of the latent interaction space.
  \item $K_1$: number of ordinal categories in Layer 4; $K_2$: number of ordinal categories in Layer 5.
  \item $\bm{a}_i \in \mathbb{R}^d$: shared latent position of respondent $i$ (common across all layers).
  \item $\bm{b}^{(l)}_j \in \mathbb{R}^d$: latent position of item $j$ in layer $l$.
  \item $\alpha^{(l)}_i \in \mathbb{R}$: respondent effect for respondent $i$ in layer $l$.
  \item $\beta^{(l)}_j \in \mathbb{R}$: item difficulty for item $j$ in layers $l=1,2,3$.
  \item $\gamma_l > 0$: distance weight for layer $l$; parametrized as $\gamma_l = \exp(\log\gamma_l)$.
  \item $\kappa > 0$: overdispersion parameter for the count layer; parametrized as $\kappa = \exp(\log\kappa)$.
  \item $\sigma^2_{\alpha_l}$: variance of the respondent effects in layer $l$.
  \item $\tau^2_{\beta_l}$: variance of item difficulties in layers $l=1,2,3$ (fixed at 1).
  \item $\sigma^2_0$: residual variance for the continuous layer.
  \item For ordinal layers ($l=4,5$): thresholds $c^{(l)}_{j,1} < \cdots < c^{(l)}_{j,K_l-1}$ with $c^{(l)}_{j,1} = 0$ (fixed) and increments $\delta^{(l)}_{j,k}$ for $k=1,\dots,K_l-2$ such that
  \[
    c^{(l)}_{j,k+1} = c^{(l)}_{j,k} + \mathrm{softplus}(\delta^{(l)}_{j,k}), \qquad \mathrm{softplus}(x) = \log(1+e^x).
  \]
\end{itemize}

%% ============================================================
\section{Data and Likelihood}
%% ============================================================

\subsection{Layer 1 --- Binary (Bernoulli--Logit)}

Data: $Y^{(1)}_{ij} \in \{0,1\}$, \; $i=1,\dots,n$, \; $j=1,\dots,P_1$.

Linear predictor:
\[
  \eta^{(1)}_{ij} = \alpha^{(1)}_i - \beta^{(1)}_j - \gamma_1 \|\bm{a}_i - \bm{b}^{(1)}_j\|.
\]
Likelihood:
\[
  Y^{(1)}_{ij} \mid \eta^{(1)}_{ij} \ind \mathrm{Bernoulli}\!\left(\mathrm{logit}^{-1}(\eta^{(1)}_{ij})\right),
\]
i.e.\ $\log p(Y^{(1)}_{ij}\mid\cdot) = Y^{(1)}_{ij}\,\eta^{(1)}_{ij} - \log(1+e^{\eta^{(1)}_{ij}})$.

\subsection{Layer 2 --- Continuous (Normal)}

Data: $Y^{(2)}_{ij} \in \mathbb{R}$, \; $i=1,\dots,n$, \; $j=1,\dots,P_2$.

Mean function:
\[
  \mu^{(2)}_{ij} = \alpha^{(2)}_i - \beta^{(2)}_j - \gamma_2 \|\bm{a}_i - \bm{b}^{(2)}_j\|.
\]
Likelihood:
\[
  Y^{(2)}_{ij} \mid \mu^{(2)}_{ij}, \sigma^2_0 \ind \mathrm{N}\!\left(\mu^{(2)}_{ij},\; \sigma^2_0\right).
\]

\subsection{Layer 3 --- Count (Negative Binomial)}

Data: $Y^{(3)}_{ij} \in \{0,1,2,\dots\}$, \; $i=1,\dots,n$, \; $j=1,\dots,P_3$.

Log-mean:
\[
  \log\mu^{(3)}_{ij} = \alpha^{(3)}_i - \beta^{(3)}_j - \gamma_3 \|\bm{a}_i - \bm{b}^{(3)}_j\|.
\]
Let $r = 1/\kappa$ (size parameter). Then:
\[
  Y^{(3)}_{ij} \mid \mu^{(3)}_{ij}, \kappa \ind \mathrm{NegBin}\!\left(r,\; \frac{r}{r + \mu^{(3)}_{ij}}\right),
\]
where $\mathrm{NegBin}(r,p)$ has probability mass function
$\binom{y+r-1}{y}\,p^r\,(1-p)^y$.

\subsection{Layer 4 --- Ordinal 1 (Cumulative Logit)}

Data: $Y^{(4)}_{ij} \in \{1,\dots,K_1\}$, \; $i=1,\dots,n$, \; $j=1,\dots,P_4$.

Linear predictor (no $\beta$):
\[
  \eta^{(4)}_{ij} = \alpha^{(4)}_i - \gamma_4 \|\bm{a}_i - \bm{b}^{(4)}_j\|.
\]
Cumulative probabilities:
\[
  C^{(4)}_{ij,k} = \mathrm{logit}^{-1}\!\left(\eta^{(4)}_{ij} - c^{(4)}_{j,k}\right), \quad k=1,\dots,K_1-1.
\]
Category probabilities:
\[
  P\!\left(Y^{(4)}_{ij} = k \mid \cdot\right) =
  \begin{cases}
    1 - C^{(4)}_{ij,1}, & k = 1, \\
    C^{(4)}_{ij,k-1} - C^{(4)}_{ij,k}, & 2 \le k \le K_1-1, \\
    C^{(4)}_{ij,K_1-1}, & k = K_1.
  \end{cases}
\]

\subsection{Layer 5 --- Ordinal 2 (Cumulative Logit)}

Data: $Y^{(5)}_{ij} \in \{1,\dots,K_2\}$, \; $i=1,\dots,n$, \; $j=1,\dots,P_5$.

Identical structure to Layer 4 with its own parameters:
\[
  \eta^{(5)}_{ij} = \alpha^{(5)}_i - \gamma_5 \|\bm{a}_i - \bm{b}^{(5)}_j\|,
\]
and thresholds $c^{(5)}_{j,k}$ constructed from $\delta^{(5)}_{j,k}$ for $k=1,\dots,K_2-2$.

%% ============================================================
\section{Prior Distributions}
%% ============================================================

\subsection{Respondent Effects}

\[
  \alpha^{(l)}_i \mid \sigma^2_{\alpha_l} \iid \mathrm{N}(0,\; \sigma^2_{\alpha_l}), \quad i=1,\dots,n, \quad l=1,\dots,5.
\]

\subsection{Item Difficulties (Layers 1, 2, 3)}

\[
  \beta^{(l)}_j \mid \tau^2_{\beta_l} \iid \mathrm{N}(0,\; \tau^2_{\beta_l}), \quad j=1,\dots,P_l, \quad l=1,2,3.
\]
In this implementation $\tau^2_{\beta_l} \equiv 1$ (fixed).

\subsection{Latent Positions}

Shared respondent positions:
\[
  \bm{a}_i \iid \mathrm{N}_d(\bm{0},\; \bm{I}_d), \quad i=1,\dots,n.
\]
Item positions (per layer):
\[
  \bm{b}^{(l)}_j \iid \mathrm{N}_d(\bm{0},\; \bm{I}_d), \quad j=1,\dots,P_l, \quad l=1,\dots,5.
\]

\subsection{Distance Weights}

\[
  \log\gamma_l \iid \mathrm{N}(\mu_{\log\gamma_l},\; \sigma^2_{\log\gamma_l}), \quad l=1,\dots,5.
\]

\subsection{Overdispersion (Count Layer)}

\[
  \log\kappa \sim \mathrm{N}(\mu_{\log\kappa},\; \sigma^2_{\log\kappa}).
\]

\subsection{Variance Parameters}

\[
  \sigma^2_{\alpha_l} \sim \mathrm{Inv\text{-}Gamma}(a_\sigma,\; b_\sigma), \quad l=1,\dots,5,
\]
\[
  \sigma^2_0 \sim \mathrm{Inv\text{-}Gamma}(a_{\sigma_0},\; b_{\sigma_0}).
\]

\subsection{Ordinal Threshold Increments}

For Layer 4 ($K_1 > 2$):
\[
  \delta^{(4)}_{j,k} \iid \mathrm{N}(\mu_\delta,\; \sigma^2_\delta), \quad j=1,\dots,P_4,\; k=1,\dots,K_1-2.
\]
For Layer 5 ($K_2 > 2$):
\[
  \delta^{(5)}_{j,k} \iid \mathrm{N}(\mu_\delta,\; \sigma^2_\delta), \quad j=1,\dots,P_5,\; k=1,\dots,K_2-2.
\]

%% ============================================================
\section{MCMC Sampling Algorithm}
%% ============================================================

All Metropolis--Hastings (MH) steps use a symmetric random-walk proposal
$\theta^* \sim \mathrm{N}(\theta^{(\mathrm{cur})}, \sigma^2_{\mathrm{prop}})$
with acceptance ratio $\min\!\big(1,\; \frac{p(\theta^*\mid\cdot)}{p(\theta^{(\mathrm{cur})}\mid\cdot)}\big)$.
Gibbs steps sample directly from the full conditional.

\begin{algorithm}[H]
\SetAlgoLined
\caption{MCMC for 5-Layered Nonhierarchical LSIRM}
\For{$\mathrm{iter} = 1, \dots, T$}{

\s
\textbf{Step 1. Update respondent effects $\alpha^{(l)}_i$ (MH, per layer)} \\
\For{$l = 1, \dots, 5$}{
  \For{$i = 1, \dots, n$}{
    Propose $\alpha^{(l)*}_i \sim \mathrm{N}(\alpha^{(l)}_i, \sigma^2_{\mathrm{prop},\alpha_l})$. \\
    Log-prior ratio: $\displaystyle \frac{-(\alpha^{(l)*}_i)^2 + (\alpha^{(l)}_i)^2}{2\sigma^2_{\alpha_l}}$. \\
    Log-likelihood ratio: $\displaystyle \sum_{j=1}^{P_l}\left[\ell_l(Y^{(l)}_{ij} \mid \alpha^{(l)*}_i) - \ell_l(Y^{(l)}_{ij} \mid \alpha^{(l)}_i)\right]$. \\
    Accept/reject by MH ratio.
  }
}

\s
\textbf{Step 2. Update item difficulties $\beta^{(l)}_j$ for $l=1,2,3$ (MH)} \\
\For{$l \in \{1, 2, 3\}$}{
  \For{$j = 1, \dots, P_l$}{
    Propose $\beta^{(l)*}_j \sim \mathrm{N}(\beta^{(l)}_j, \sigma^2_{\mathrm{prop},\beta_l})$. \\
    Log-prior ratio: $\displaystyle \frac{-(\beta^{(l)*}_j)^2 + (\beta^{(l)}_j)^2}{2\tau^2_{\beta_l}}$. \\
    Log-likelihood ratio: $\displaystyle \sum_{i=1}^{n}\left[\ell_l(Y^{(l)}_{ij} \mid \beta^{(l)*}_j) - \ell_l(Y^{(l)}_{ij} \mid \beta^{(l)}_j)\right]$. \\
    Accept/reject by MH ratio.
  }
}

\s
\textbf{Step 3. Update distance weights $\log\gamma_l$ for $l=1,\dots,5$ (MH)} \\
\For{$l = 1, \dots, 5$}{
  Propose $\log\gamma^*_l \sim \mathrm{N}(\log\gamma_l, \sigma^2_{\mathrm{prop},\log\gamma_l})$, set $\gamma^*_l = \exp(\log\gamma^*_l)$. \\
  Log-prior ratio: $\displaystyle \frac{-(\log\gamma^*_l - \mu_{\log\gamma_l})^2 + (\log\gamma_l - \mu_{\log\gamma_l})^2}{2\sigma^2_{\log\gamma_l}}$. \\
  Log-likelihood ratio: over all $(i,j)$ pairs in layer $l$. \\
  Accept/reject by MH ratio.
}

\s
\textbf{Step 4. Update shared respondent positions $\bm{a}_i$ (MH)} \\
\For{$i = 1, \dots, n$}{
  Propose $\bm{a}^*_i \sim \mathrm{N}_d(\bm{a}_i, \sigma^2_{\mathrm{prop},a}\,\bm{I}_d)$. \\
  Log-prior ratio: $\displaystyle \frac{-\|\bm{a}^*_i\|^2 + \|\bm{a}_i\|^2}{2}$. \\
  Log-likelihood ratio: sum over \textbf{all five layers}:
  \[
    \sum_{l=1}^{5}\sum_{j=1}^{P_l}\left[\ell_l(Y^{(l)}_{ij} \mid \bm{a}^*_i) - \ell_l(Y^{(l)}_{ij} \mid \bm{a}_i)\right].
  \]
  Accept/reject by MH ratio.
}

\s
\textbf{Step 5. Update item positions $\bm{b}^{(l)}_j$ for $l=1,\dots,5$ (MH)} \\
\For{$l = 1, \dots, 5$}{
  \For{$j = 1, \dots, P_l$}{
    Propose $\bm{b}^{(l)*}_j \sim \mathrm{N}_d(\bm{b}^{(l)}_j, \sigma^2_{\mathrm{prop},b_l}\,\bm{I}_d)$. \\
    Log-prior ratio: $\displaystyle \frac{-\|\bm{b}^{(l)*}_j\|^2 + \|\bm{b}^{(l)}_j\|^2}{2}$. \\
    Log-likelihood ratio: $\displaystyle \sum_{i=1}^{n}\left[\ell_l(Y^{(l)}_{ij} \mid \bm{b}^{(l)*}_j) - \ell_l(Y^{(l)}_{ij} \mid \bm{b}^{(l)}_j)\right]$. \\
    Accept/reject by MH ratio.
  }
}

\s
\textbf{Step 6a. Update ordinal thresholds for Layer 4 (MH)} \\
\For{$j = 1, \dots, P_4$}{
  Propose $\delta^{(4)*}_{j,k} \sim \mathrm{N}(\delta^{(4)}_{j,k}, \sigma^2_{\mathrm{prop},\delta})$ for $k=1,\dots,K_1-2$. \\
  Reconstruct proposed thresholds: $c^{(4)*}_{j,1} = 0$, \;
  $c^{(4)*}_{j,k+1} = c^{(4)*}_{j,k} + \mathrm{softplus}(\delta^{(4)*}_{j,k})$. \\
  Log-prior ratio: $\displaystyle \sum_{k=1}^{K_1-2} \frac{-(\delta^{(4)*}_{j,k}-\mu_\delta)^2 + (\delta^{(4)}_{j,k}-\mu_\delta)^2}{2\sigma^2_\delta}$. \\
  Jacobian correction: $\displaystyle \sum_{k=1}^{K_1-2} \left[\log\sigma(\delta^{(4)*}_{j,k}) - \log\sigma(\delta^{(4)}_{j,k})\right]$,\; where $\sigma(\cdot)$ is the sigmoid. \\
  Log-likelihood ratio: $\displaystyle \sum_{i=1}^{n}\left[\log P(Y^{(4)}_{ij}\mid \bm{c}^{(4)*}_j) - \log P(Y^{(4)}_{ij}\mid \bm{c}^{(4)}_j)\right]$. \\
  Accept/reject by MH ratio (likelihood $\times$ prior $\times$ Jacobian).
}

\s
\textbf{Step 6b. Update ordinal thresholds for Layer 5 (MH)} \\
Identical to Step 6a with $P_5$, $K_2$, and $\delta^{(5)}_{j,k}$.

\s
\textbf{Step 7. Update overdispersion $\log\kappa$ (MH)} \\
Propose $\log\kappa^* \sim \mathrm{N}(\log\kappa, \sigma^2_{\mathrm{prop},\log\kappa})$, set $\kappa^* = \exp(\log\kappa^*)$. \\
Log-prior ratio: $\displaystyle \frac{-(\log\kappa^* - \mu_{\log\kappa})^2 + (\log\kappa - \mu_{\log\kappa})^2}{2\sigma^2_{\log\kappa}}$. \\
Log-likelihood ratio: over all $(i,j)$ in layer 3 with $r^* = 1/\kappa^*$. \\
Accept/reject by MH ratio.

\s
\textbf{Step 8. Gibbs updates for variance parameters} \\
\begin{enumerate}
  \item[(a)] Residual variance (continuous layer):
  \[
    \sigma^2_0 \mid \cdot \;\sim\; \mathrm{Inv\text{-}Gamma}\!\left(a_{\sigma_0} + \frac{nP_2}{2},\;\; b_{\sigma_0} + \frac{1}{2}\sum_{i=1}^{n}\sum_{j=1}^{P_2}\left(Y^{(2)}_{ij} - \mu^{(2)}_{ij}\right)^2\right).
  \]
  \item[(b)] Respondent-effect variances ($l=1,\dots,5$):
  \[
    \sigma^2_{\alpha_l} \mid \cdot \;\sim\; \mathrm{Inv\text{-}Gamma}\!\left(a_\sigma + \frac{n}{2},\;\; b_\sigma + \frac{1}{2}\sum_{i=1}^{n}(\alpha^{(l)}_i)^2\right).
  \]
\end{enumerate}

\s
\textbf{Step 9. Store samples} (after burn-in, with thinning).
}
\end{algorithm}

%% ============================================================
\section{Summary of Layer-Specific Log-Likelihoods $\ell_l$}
%% ============================================================

For reference, the per-observation log-likelihoods used in the MH ratios above are:

\begin{align}
  \ell_1(Y^{(1)}_{ij}\mid\cdot) &= Y^{(1)}_{ij}\,\eta^{(1)}_{ij} - \log(1+e^{\eta^{(1)}_{ij}}), \label{eq:ll1}\\[6pt]
  \ell_2(Y^{(2)}_{ij}\mid\cdot) &= -\frac{1}{2}\log(2\pi\sigma^2_0) - \frac{(Y^{(2)}_{ij}-\mu^{(2)}_{ij})^2}{2\sigma^2_0}, \label{eq:ll2}\\[6pt]
  \ell_3(Y^{(3)}_{ij}\mid\cdot) &= \log\binom{Y^{(3)}_{ij}+r-1}{Y^{(3)}_{ij}} + r\log\frac{r}{r+\mu^{(3)}_{ij}} + Y^{(3)}_{ij}\log\frac{\mu^{(3)}_{ij}}{r+\mu^{(3)}_{ij}}, \label{eq:ll3}\\[6pt]
  \ell_4(Y^{(4)}_{ij}\mid\cdot) &= \log P(Y^{(4)}_{ij} = k \mid \eta^{(4)}_{ij}, \bm{c}^{(4)}_j), \label{eq:ll4}\\[6pt]
  \ell_5(Y^{(5)}_{ij}\mid\cdot) &= \log P(Y^{(5)}_{ij} = k \mid \eta^{(5)}_{ij}, \bm{c}^{(5)}_j), \label{eq:ll5}
\end{align}
where $r = 1/\kappa$ and $\mu^{(3)}_{ij} = \exp(\alpha^{(3)}_i - \beta^{(3)}_j - \gamma_3\|\bm{a}_i - \bm{b}^{(3)}_j\|)$.
