\section{Notation}

\begin{itemize}[leftmargin=1.5em]
    \item $n$: number of respondents (subjects), indexed by $i=1,\dots,n$.
    \item $P_l$: number of items in layer $l$ ($l=1,\dots,5$).
    \item $d$: dimension of the latent interaction space.
    \item $K_1$: number of ordinal categories in Layer 4.
    \item $K_2$: number of ordinal categories in Layer 5.
    \item $a_i \in \mathbb{R}^d$: shared latent position of respondent $i$ (common across all layers).
    \item $b_j^{(l)} \in \mathbb{R}^d$: latent position of item $j$ in layer $l$.
    \item $\alpha_i^{(l)} \in \mathbb{R}$: respondent effect for respondent $i$ in layer $l$.
    \item $\beta_j^{(l)} \in \mathbb{R}$: item difficulty for item $j$ in layers $l=1,2,3$.
    \item $\gamma_l > 0$: distance weight for layer $l$, parametrized as $\gamma_l = \exp(\log \gamma_l)$.
    \item $\kappa > 0$: overdispersion parameter for the count layer, parametrized as $\kappa = \exp(\log \kappa)$.
    \item $\sigma_{\alpha l}^2$: variance of the respondent effects in layer $l$.
    \item $\tau_{\beta l}^2$: variance of item difficulties in layers $l=1,2,3$ (fixed at $1$).
    \item $\sigma_0^2$: residual scale parameter for the continuous layer.
    \item $\nu_2 > 0$: degrees of freedom for the robust continuous layer. In the present formulation, $\nu_2$ is treated as fixed.
    \item $\lambda_{ij}^{(2)} > 0$: observation-specific latent scale variable for Layer 2, introduced through the normal--gamma mixture representation of the Student-$t$ distribution.
    \item For ordinal layers ($l=4,5$): thresholds $c_{j,1}^{(l)} < \cdots < c_{j,K_l-1}^{(l)}$ with $c_{j,1}^{(l)} = 0$ (fixed) and increments $\delta_{j,k}^{(l)}$ for $k=1,\dots,K_l-2$ such that
    \[
    c_{j,k+1}^{(l)} = c_{j,k}^{(l)} + \operatorname{softplus}(\delta_{j,k}^{(l)}),
    \qquad
    \operatorname{softplus}(x)=\log(1+e^x).
    \]
\end{itemize}

\section{Data and Likelihood}

\subsection{Layer 1 --- Binary (Bernoulli--Logit)}

Data: $Y_{ij}^{(1)} \in \{0,1\}$, $i=1,\dots,n$, $j=1,\dots,P_1$.

Linear predictor:
\[
\eta_{ij}^{(1)} = \alpha_i^{(1)} - \beta_j^{(1)} - \gamma_1\|a_i - b_j^{(1)}\|.
\]

Likelihood:
\[
Y_{ij}^{(1)} \mid \eta_{ij}^{(1)} \overset{ind}{\sim} \operatorname{Bernoulli}\!\left(\operatorname{logit}^{-1}(\eta_{ij}^{(1)})\right),
\]
that is,
\[
\log p\bigl(Y_{ij}^{(1)}\mid \cdot\bigr) = Y_{ij}^{(1)}\eta_{ij}^{(1)} - \log\bigl(1+e^{\eta_{ij}^{(1)}}\bigr).
\]

\subsection{Layer 2 --- Continuous (Robust Student-$t$ via Normal--Gamma Mixture)}

Data: $Y_{ij}^{(2)} \in \mathbb{R}$, $i=1,\dots,n$, $j=1,\dots,P_2$.

Mean function:
\[
\mu_{ij}^{(2)} = \alpha_i^{(2)} - \beta_j^{(2)} - \gamma_2\|a_i - b_j^{(2)}\|.
\]

To keep the implementation as close as possible to the original code, the structural part of the model is unchanged, and robustness is introduced only through an observation-specific latent weight $\lambda_{ij}^{(2)}$. Specifically,
\begin{align*}
Y_{ij}^{(2)} \mid \mu_{ij}^{(2)}, \sigma_0^2, \lambda_{ij}^{(2)}
&\overset{ind}{\sim}
N\!\left(\mu_{ij}^{(2)}, \frac{\sigma_0^2}{\lambda_{ij}^{(2)}}\right),\\
\lambda_{ij}^{(2)} \mid \nu_2
&\overset{iid}{\sim}
\operatorname{Gamma}\!\left(\frac{\nu_2}{2},\frac{\nu_2}{2}\right).
\end{align*}

Marginally, this implies
\[
Y_{ij}^{(2)} \mid \mu_{ij}^{(2)},\sigma_0^2,\nu_2
\sim t_{\nu_2}\!\left(\mu_{ij}^{(2)},\sigma_0^2\right),
\]
so Layer 2 becomes a robust continuous layer with Student-$t$ errors. The key distinction from the original specification is the introduction of $\lambda_{ij}^{(2)}$; all remaining model components are kept aligned with the original code structure.

Conditional log-likelihood given $\lambda_{ij}^{(2)}$:
\[
\log p\bigl(Y_{ij}^{(2)} \mid \cdot\bigr)
=
-\frac{1}{2}\log\left(2\pi \frac{\sigma_0^2}{\lambda_{ij}^{(2)}}\right)
-
\frac{\lambda_{ij}^{(2)}}{2\sigma_0^2}\left(Y_{ij}^{(2)}-\mu_{ij}^{(2)}\right)^2.
\]

\subsection{Layer 3 --- Count (Negative Binomial)}

Data: $Y_{ij}^{(3)} \in \{0,1,2,\dots\}$, $i=1,\dots,n$, $j=1,\dots,P_3$.

Log-mean:
\[
\log \mu_{ij}^{(3)} = \alpha_i^{(3)} - \beta_j^{(3)} - \gamma_3\|a_i - b_j^{(3)}\|.
\]

Let $r=1/\kappa$ (size parameter). Then
\[
Y_{ij}^{(3)} \mid \mu_{ij}^{(3)},\kappa
\overset{ind}{\sim}
\operatorname{NegBin}\left(r, \frac{r}{r+\mu_{ij}^{(3)}}\right),
\]
where $\operatorname{NegBin}(r,p)$ has probability mass function
\[
\Pr(Y=y)=\binom{y+r-1}{y}p^r(1-p)^y.
\]

\subsection{Layer 4 --- Ordinal 1 (Cumulative Logit)}

Data: $Y_{ij}^{(4)} \in \{1,\dots,K_1\}$, $i=1,\dots,n$, $j=1,\dots,P_4$.

Linear predictor (no $\beta$):
\[
\eta_{ij}^{(4)} = \alpha_i^{(4)} - \gamma_4\|a_i - b_j^{(4)}\|.
\]

Cumulative probabilities:
\[
C_{ij,k}^{(4)} = \operatorname{logit}^{-1}\!\left(\eta_{ij}^{(4)} - c_{j,k}^{(4)}\right),
\qquad k=1,\dots,K_1-1.
\]

Category probabilities:
\[
\Pr\bigl(Y_{ij}^{(4)}=k\mid \cdot\bigr)=
\begin{cases}
1-C_{ij,1}^{(4)}, & k=1,\\
C_{ij,k-1}^{(4)}-C_{ij,k}^{(4)}, & 2\le k\le K_1-1,\\
C_{ij,K_1-1}^{(4)}, & k=K_1.
\end{cases}
\]

\subsection{Layer 5 --- Ordinal 2 (Cumulative Logit)}

Data: $Y_{ij}^{(5)} \in \{1,\dots,K_2\}$, $i=1,\dots,n$, $j=1,\dots,P_5$.

The structure is identical to Layer 4 with its own parameters:
\[
\eta_{ij}^{(5)} = \alpha_i^{(5)} - \gamma_5\|a_i - b_j^{(5)}\|,
\]
and thresholds $c_{j,k}^{(5)}$ constructed from $\delta_{j,k}^{(5)}$ for $k=1,\dots,K_2-2$.

\section{Prior Distributions}

\subsection{Respondent Effects}
\[
\alpha_i^{(l)} \mid \sigma_{\alpha l}^2 \overset{iid}{\sim} N(0,\sigma_{\alpha l}^2),
\qquad i=1,\dots,n,\; l=1,\dots,5.
\]

\subsection{Item Difficulties (Layers 1, 2, 3)}
\[
\beta_j^{(l)} \mid \tau_{\beta l}^2 \overset{iid}{\sim} N(0,\tau_{\beta l}^2),
\qquad j=1,\dots,P_l,\; l=1,2,3.
\]
In this implementation, $\tau_{\beta l}^2 \equiv 1$ is fixed.

\subsection{Latent Positions}
Shared respondent positions:
\[
a_i \overset{iid}{\sim} N_d(0,I_d),
\qquad i=1,\dots,n.
\]
Item positions (per layer):
\[
b_j^{(l)} \overset{iid}{\sim} N_d(0,I_d),
\qquad j=1,\dots,P_l,\; l=1,\dots,5.
\]

\subsection{Distance Weights}
\[
\log \gamma_l \overset{iid}{\sim} N(\mu_{\log \gamma_l},\sigma_{\log \gamma_l}^2),
\qquad l=1,\dots,5.
\]

\subsection{Overdispersion (Count Layer)}
\[
\log \kappa \sim N(\mu_{\log \kappa},\sigma_{\log \kappa}^2).
\]

\subsection{Variance Parameters}
\[
\sigma_{\alpha l}^2 \sim \operatorname{Inv\text{-}Gamma}(a_\sigma,b_\sigma),
\qquad l=1,\dots,5,
\]
\[
\sigma_0^2 \sim \operatorname{Inv\text{-}Gamma}(a_{\sigma_0},b_{\sigma_0}).
\]

\subsection{Ordinal Threshold Increments}
For Layer 4 ($K_1>2$):
\[
\delta_{j,k}^{(4)} \overset{iid}{\sim} N(\mu_\delta,\sigma_\delta^2),
\qquad j=1,\dots,P_4,\; k=1,\dots,K_1-2.
\]
For Layer 5 ($K_2>2$):
\[
\delta_{j,k}^{(5)} \overset{iid}{\sim} N(\mu_\delta,\sigma_\delta^2),
\qquad j=1,\dots,P_5,\; k=1,\dots,K_2-2.
\]

\section{MCMC Sampling Algorithm}

All Metropolis--Hastings (MH) steps use a symmetric random-walk proposal $\theta^*\sim N(\theta^{(cur)},\sigma_{prop}^2)$ with acceptance ratio
\[
\min\left\{1,\frac{p(\theta^*\mid \cdot)}{p(\theta^{(cur)}\mid \cdot)}\right\}.
\]
Gibbs steps sample directly from the full conditional distribution.

\bigskip
\noindent
\textbf{Algorithm 1: MCMC for 5-Layered Nonhierarchical LSIRM with Robust Continuous Layer}

\begin{enumerate}[leftmargin=1.8em]
    \item[for] $iter=1,\dots,T$ do

    \item \textbf{Step 0. Update latent scales $\lambda_{ij}^{(2)}$ for Layer 2 (Gibbs).}
    For each $i=1,\dots,n$ and $j=1,\dots,P_2$, let
    \[
    e_{ij}^{(2)} = Y_{ij}^{(2)} - \mu_{ij}^{(2)}.
    \]
    Then
    \[
    \lambda_{ij}^{(2)}\mid \cdot
    \sim
    \operatorname{Gamma}\left(
    \frac{\nu_2+1}{2},
    \frac{\nu_2 + \{e_{ij}^{(2)}\}^2/\sigma_0^2}{2}
    \right).
    \]

    \item \textbf{Step 1. Update respondent effects $\alpha_i^{(l)}$ (MH, per layer).}
    For $l=1,\dots,5$, and for $i=1,\dots,n$:
    \begin{itemize}[leftmargin=1.5em]
        \item Propose $\alpha_i^{(l)*}\sim N(\alpha_i^{(l)},\sigma_{prop,\alpha l}^2)$.
        \item Log-prior ratio:
        \[
        -\frac{(\alpha_i^{(l)*})^2-(\alpha_i^{(l)})^2}{2\sigma_{\alpha l}^2}.
        \]
        \item Log-likelihood ratio:
        \[
        \sum_{j=1}^{P_l}\left[\ell_l\bigl(Y_{ij}^{(l)}\mid \alpha_i^{(l)*}\bigr)-\ell_l\bigl(Y_{ij}^{(l)}\mid \alpha_i^{(l)}\bigr)\right].
        \]
        \item Accept/reject by the MH ratio.
    \end{itemize}

    \item \textbf{Step 2. Update item difficulties $\beta_j^{(l)}$ for $l=1,2,3$ (MH).}
    For $l\in\{1,2,3\}$ and for $j=1,\dots,P_l$:
    \begin{itemize}[leftmargin=1.5em]
        \item Propose $\beta_j^{(l)*}\sim N(\beta_j^{(l)},\sigma_{prop,\beta l}^2)$.
        \item Log-prior ratio:
        \[
        -\frac{(\beta_j^{(l)*})^2-(\beta_j^{(l)})^2}{2\tau_{\beta l}^2}.
        \]
        \item Log-likelihood ratio:
        \[
        \sum_{i=1}^{n}\left[\ell_l\bigl(Y_{ij}^{(l)}\mid \beta_j^{(l)*}\bigr)-\ell_l\bigl(Y_{ij}^{(l)}\mid \beta_j^{(l)}\bigr)\right].
        \]
        \item Accept/reject by the MH ratio.
    \end{itemize}

    \item \textbf{Step 3. Update distance weights $\log \gamma_l$ for $l=1,\dots,5$ (MH).}
    For $l=1,\dots,5$:
    \begin{itemize}[leftmargin=1.5em]
        \item Propose $\log \gamma_l^*\sim N(\log \gamma_l,\sigma_{prop,\log \gamma_l}^2)$, and set $\gamma_l^*=\exp(\log \gamma_l^*)$.
        \item Log-prior ratio:
        \[
        -\frac{(\log \gamma_l^*-\mu_{\log \gamma_l})^2-(\log \gamma_l-\mu_{\log \gamma_l})^2}{2\sigma_{\log \gamma_l}^2}.
        \]
        \item Log-likelihood ratio: sum over all $(i,j)$ pairs in layer $l$.
        \item Accept/reject by the MH ratio.
    \end{itemize}

    \item \textbf{Step 4. Update shared respondent positions $a_i$ (MH).}
    For $i=1,\dots,n$:
    \begin{itemize}[leftmargin=1.5em]
        \item Propose $a_i^*\sim N_d(a_i,\sigma_{prop,a}^2 I_d)$.
        \item Log-prior ratio:
        \[
        -\frac{\|a_i^*\|^2-\|a_i\|^2}{2}.
        \]
        \item Log-likelihood ratio:
        \[
        \sum_{l=1}^{5}\sum_{j=1}^{P_l}
        \left[\ell_l\bigl(Y_{ij}^{(l)}\mid a_i^*\bigr)-\ell_l\bigl(Y_{ij}^{(l)}\mid a_i\bigr)\right].
        \]
        \item Accept/reject by the MH ratio.
    \end{itemize}

    \item \textbf{Step 5. Update item positions $b_j^{(l)}$ for $l=1,\dots,5$ (MH).}
    For $l=1,\dots,5$ and $j=1,\dots,P_l$:
    \begin{itemize}[leftmargin=1.5em]
        \item Propose $b_j^{(l)*}\sim N_d(b_j^{(l)},\sigma_{prop,b_l}^2 I_d)$.
        \item Log-prior ratio:
        \[
        -\frac{\|b_j^{(l)*}\|^2-\|b_j^{(l)}\|^2}{2}.
        \]
        \item Log-likelihood ratio:
        \[
        \sum_{i=1}^{n}
        \left[\ell_l\bigl(Y_{ij}^{(l)}\mid b_j^{(l)*}\bigr)-\ell_l\bigl(Y_{ij}^{(l)}\mid b_j^{(l)}\bigr)\right].
        \]
        \item Accept/reject by the MH ratio.
    \end{itemize}

    \item \textbf{Step 6a. Update ordinal thresholds for Layer 4 (MH).}
    For $j=1,\dots,P_4$:
    \begin{itemize}[leftmargin=1.5em]
        \item Propose $\delta_{j,k}^{(4)*}\sim N(\delta_{j,k}^{(4)},\sigma_{prop,\delta}^2)$ for $k=1,\dots,K_1-2$.
        \item Reconstruct proposed thresholds: $c_{j,1}^{(4)*}=0$ and
        \[
        c_{j,k+1}^{(4)*}=c_{j,k}^{(4)*}+\operatorname{softplus}(\delta_{j,k}^{(4)*}).
        \]
        \item Log-prior ratio:
        \[
        \sum_{k=1}^{K_1-2}
        -\frac{(\delta_{j,k}^{(4)*}-\mu_\delta)^2-(\delta_{j,k}^{(4)}-\mu_\delta)^2}{2\sigma_\delta^2}.
        \]
        \item Jacobian correction:
        \[
        \sum_{k=1}^{K_1-2}\bigl[\log s(\delta_{j,k}^{(4)*})-\log s(\delta_{j,k}^{(4)})\bigr],
        \]
        where $s(\cdot)$ denotes the sigmoid function.
        \item Log-likelihood ratio:
        \[
        \sum_{i=1}^{n}\left[\log \Pr\bigl(Y_{ij}^{(4)}\mid c_j^{(4)*}\bigr)-\log \Pr\bigl(Y_{ij}^{(4)}\mid c_j^{(4)}\bigr)\right].
        \]
        \item Accept/reject by the MH ratio.
    \end{itemize}

    \item \textbf{Step 6b. Update ordinal thresholds for Layer 5 (MH).}
    This step is identical to Step 6a with $P_5$, $K_2$, and $\delta_{j,k}^{(5)}$ replacing $P_4$, $K_1$, and $\delta_{j,k}^{(4)}$, respectively.

    \item \textbf{Step 7. Update overdispersion $\log \kappa$ (MH).}
    \begin{itemize}[leftmargin=1.5em]
        \item Propose $\log \kappa^*\sim N(\log \kappa,\sigma_{prop,\log \kappa}^2)$ and set $\kappa^*=\exp(\log \kappa^*)$.
        \item Log-prior ratio:
        \[
        -\frac{(\log \kappa^*-\mu_{\log \kappa})^2-(\log \kappa-\mu_{\log \kappa})^2}{2\sigma_{\log \kappa}^2}.
        \]
        \item Log-likelihood ratio: over all $(i,j)$ in Layer 3 with $r^*=1/\kappa^*$.
        \item Accept/reject by the MH ratio.
    \end{itemize}

    \item \textbf{Step 8. Gibbs updates for variance parameters.}

    \begin{itemize}[leftmargin=1.5em]
        \item[(a)] \textit{Residual variance for the robust continuous layer:}
        \[
        \sigma_0^2\mid \cdot
        \sim
        \operatorname{Inv\text{-}Gamma}
        \left(
        a_{\sigma_0}+\frac{nP_2}{2},
        b_{\sigma_0}+\frac{1}{2}\sum_{i=1}^{n}\sum_{j=1}^{P_2}\lambda_{ij}^{(2)}\bigl(Y_{ij}^{(2)}-\mu_{ij}^{(2)}\bigr)^2
        \right).
        \]
        Relative to the original Gaussian Layer 2 update, the only modification is that the residual sum of squares is replaced by the weighted residual sum of squares with weights $\lambda_{ij}^{(2)}$.

        \item[(b)] \textit{Respondent-effect variances} ($l=1,\dots,5$):
        \[
        \sigma_{\alpha l}^2\mid \cdot
        \sim
        \operatorname{Inv\text{-}Gamma}
        \left(
        a_\sigma+\frac{n}{2},
        b_\sigma+\frac{1}{2}\sum_{i=1}^{n}(\alpha_i^{(l)})^2
        \right).
        \]
    \end{itemize}

    \item \textbf{Step 9. Store samples} (after burn-in and with thinning, if desired).

    \item[end] do
\end{enumerate}

\section{Summary of Layer-Specific Log-Likelihoods $\ell_l$}

For reference, the per-observation log-likelihood contributions used in the MH ratios are:
\begin{align*}
\ell_1\bigl(Y_{ij}^{(1)}\mid \cdot\bigr)
&= Y_{ij}^{(1)}\eta_{ij}^{(1)} - \log\bigl(1+e^{\eta_{ij}^{(1)}}\bigr), \\
\ell_2\bigl(Y_{ij}^{(2)}\mid \cdot,\lambda_{ij}^{(2)}\bigr)
&= -\frac{1}{2}\log\left(2\pi \frac{\sigma_0^2}{\lambda_{ij}^{(2)}}\right)
-\frac{\lambda_{ij}^{(2)}}{2\sigma_0^2}\left(Y_{ij}^{(2)}-\mu_{ij}^{(2)}\right)^2, \\
\ell_3\bigl(Y_{ij}^{(3)}\mid \cdot\bigr)
&= \log\binom{Y_{ij}^{(3)}+r-1}{Y_{ij}^{(3)}}
+ r\log\left(\frac{r}{r+\mu_{ij}^{(3)}}\right)
+ Y_{ij}^{(3)}\log\left(\frac{\mu_{ij}^{(3)}}{r+\mu_{ij}^{(3)}}\right), \\
\ell_4\bigl(Y_{ij}^{(4)}\mid \cdot\bigr)
&= \log \Pr\bigl(Y_{ij}^{(4)}=k\mid \eta_{ij}^{(4)}, c_j^{(4)}\bigr), \\
\ell_5\bigl(Y_{ij}^{(5)}\mid \cdot\bigr)
&= \log \Pr\bigl(Y_{ij}^{(5)}=k\mid \eta_{ij}^{(5)}, c_j^{(5)}\bigr),
\end{align*}
where $r=1/\kappa$ and
\[
\mu_{ij}^{(3)} = \exp\left(\alpha_i^{(3)} - \beta_j^{(3)} - \gamma_3\|a_i-b_j^{(3)}\|\right).
\]

\section{Remarks on the Modification}

The original model description uses a Gaussian likelihood for Layer 2 together with the common LSIRM mean structure $\mu_{ij}^{(2)} = \alpha_i^{(2)} - \beta_j^{(2)} - \gamma_2\|a_i-b_j^{(2)}\|$. In the present robustified version, this structural component is unchanged. The only essential addition is the latent scale variable $\lambda_{ij}^{(2)}$, which yields a Student-$t$ error model after marginalization. Consequently,
\begin{itemize}[leftmargin=1.5em]
    \item the prior settings and overall code structure remain aligned with the original implementation,
    \item the MH updates for $\alpha_i^{(l)}$, $\beta_j^{(l)}$, $\log \gamma_l$, $a_i$, and $b_j^{(l)}$ are preserved, and
    \item the continuous-layer residual variance update changes only through the insertion of the weight $\lambda_{ij}^{(2)}$.
\end{itemize}
This makes the robust continuous layer straightforward to integrate into the existing sampler while preserving the rest of the 5-layer LSIRM specification.
